  ##!/bin/bash

  # 将public目录中的所有文件提交到GitHub Pages的特定分支

  # 清除并重新生成Hugo站点
  hugo -d ./public

  # 进入public目录
  cd ./public

  # 初始化一个Git仓库并提交更改
  git init
  git add .
  git commit -m "Deploying to GitHub Pages"

  # 将更改推送到GitHub Pages的特定分支，例如"gh-pages"
  git push --force git@github.com:<username>/<username>.github.io.git master

  # 返回项目根目录
  cd ..
